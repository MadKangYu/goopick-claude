from datetime import datetime, timedelta, timezone
from notion_client import Client

notion = Client(auth=os.environ["NOTION_TOKEN"])

DATABASES = {
    "products": os.environ.get("NOTION_PRODUCT_DB"),
    "orders": os.environ.get("NOTION_ORDER_DB"),

def detect_changes(since_minutes: int) -> list:
    """ì§€ì •ëœ ì‹œê°„ ë‚´ ë³€ê²½ëœ í•­ëª© ê°ì§€"""
    since_time = datetime.now(timezone.utc) - timedelta(minutes=since_minutes)
    since_iso = since_time.isoformat()
    
    all_changes = []
    
    for db_name, db_id in DATABASES.items():
        if not db_id:
            continue
            
        # ë³€ê²½ëœ í•­ëª© ì¡°íšŒ
        response = notion.databases.query(
            database_id=db_id,
            filter={
                "timestamp": "last_edited_time",
                "last_edited_time": {
                    "after": since_iso
                }
            }
        )
        
        for page in response["results"]:
            change = {
                "database": db_name,
                "page_id": page["id"],
                "last_edited": page["last_edited_time"],
                "properties": extract_key_properties(page, db_name)
            }
            all_changes.append(change)
    
    return all_changes

def extract_key_properties(page: dict, db_name: str) -> dict:
    """í•µì‹¬ ì†ì„± ì¶”ì¶œ"""
    props = page.get("properties", {})
    
    if db_name == "products":
        return {
            "sku": get_text(props.get("SKU")),
            "name": get_title(props.get("ì œí’ˆëª…")),
            "price": props.get("íŒë§¤ê°€", {}).get("number"),

            "status": get_status(props.get("ì£¼ë¬¸ìƒíƒœ")),
        }

            "cs_number": get_title(props.get("ë¬¸ì˜ë²ˆí˜¸")),
            "status": get_status(props.get("ìƒíƒœ")),
        }
    return {}

def get_title(prop):
    if prop and prop.get("title"):
        return prop["title"][0]["text"]["content"] if prop["title"] else ""
    return ""

def get_text(prop):
    if prop and prop.get("rich_text"):
        return prop["rich_text"][0]["text"]["content"] if prop["rich_text"] else ""
    return ""

def get_status(prop):
    if prop and prop.get("status"):

    return ""

if __name__ == "__main__":
